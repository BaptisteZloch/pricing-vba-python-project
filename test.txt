Private Sub inserer_delta()
    Dim delta_t As Double
    delta_t = delta()
    ThisWorkbook.Sheets("parametres").Range("delta_t").Value = delta_t
End Sub

Private Function alpha() As Double
    Dim alpha As Double
    alpha = Exp(vol * Sqr(3 * delta_t))
End Function

Private Function delta() As Double
    Dim delta_t As Double
    delta_t = ((maturity - pricing_date) / nb_step) / 365
End Function
